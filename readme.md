3. Setup husky
   3.1. npm install --save-dev husky
   3.2. npx husky init
4. npm install --save-dev lint-staged - Run tasks like formatters and linters against staged git files
5. npm install --save-dev @types/node
6. npm install --save-dev typescript && npx tsc --init
7. tsconfig configuration
8. npm install -D @commitlint/cli @commitlint/config-conventional
9. npm install --save-dev eslint @eslint/js typescript typescript-eslint
10. npm install --save-dev --save-exact prettier
